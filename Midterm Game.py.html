Midterm Game.py 
1    import pygame
2    import random
3    import time
4    
5    # Window size
6    window_x = 720
7    window_y = 480
8    
9    # Colors
10   black = pygame.Color(0,0,0)
11   white = pygame.Color(255,255,255)
12   red = pygame.Color(255,0,0)
13   green = pygame.Color(0,255,0)
14   blue = pygame.Color(0,0,255)
15   yellow = pygame.Color(255,255,0)
16   
17   # Initialize pygame
18   pygame.init()
19   
20   # Game window
21   pygame.display.set_caption("Collect Coins")
22   game_window = pygame.display.set_mode((window_x, window_y))
23   
24   # FPS controller
25   fps = pygame.time.Clock()
26   
27   # Player
28   player_size = 30
29   player_x = window_x//2
30   player_y = window_y - player_size - 10
31   player_speed = 5
32   
33   # Coin
34   coin_size = 20
35   coin_x = random.randrange(0, window_x - coin_size, 20)
36   coin_y = random.randrange(0, window_y - coin_size - 50, 20)
37   
38   # Obstacle (moving freely)
39   obstacle_width = 50
40   obstacle_height = 50
41   obstacle_x = random.randrange(0, window_x - obstacle_width, 20)
42   obstacle_y = random.randrange(0, window_y - obstacle_height, 20)
43   obstacle_speed_x = 4
44   obstacle_speed_y = 3
45   
46   # Score & Lives
47   score = 0
48   lives = 3
49   font = pygame.font.SysFont('arial', 25)
50   
51   # Game state
52   game_state = "start"  # "start", "playing", "game_over"
53   
54   # Functions
55   def draw_start_screen():
56       game_window.fill(black)
57       title = font.render("Collect Coins!", True, blue)
58       instruction = font.render("Press SPACE to Start", True, red)
59       game_window.blit(title, (window_x//2 - title.get_width()//2, window_y//2 - 50))
60       game_window.blit(instruction, (window_x//2 - instruction.get_width()//2, window_y//2))
61       pygame.display.update()
62   
63   def draw_game_over():
64       game_window.fill(black)
65       over_text = font.render("Game Over!", True, red)
66       score_text = font.render("Score: " + str(score), True, white)
67       restart_text = font.render("Press R to Restart", True, red)
68       game_window.blit(over_text, (window_x//2 - over_text.get_width()//2, window_y//2 - 60))
69       game_window.blit(score_text, (window_x//2 - score_text.get_width()//2, window_y//2 - 20))
70       game_window.blit(restart_text, (window_x//2 - restart_text.get_width()//2, window_y//2 + 20))
71       pygame.display.update()
72   
73   def reset_game():
74       global player_x, player_y, coin_x, coin_y, obstacle_x, obstacle_y, score, lives
75       player_x = window_x//2
76       player_y = window_y - player_size - 10
77       coin_x = random.randrange(0, window_x - coin_size, 20)
78       coin_y = random.randrange(0, window_y - coin_size - 50, 20)
79       obstacle_x = random.randrange(0, window_x - obstacle_width, 20)
80       obstacle_y = random.randrange(0, window_y - obstacle_height, 20)
81       score = 0
82       lives = 3
83   
84   # Main loop
85   run = True
86   while run:
87       fps.tick(60)
88       keys = pygame.key.get_pressed()
89   
90       for event in pygame.event.get():
91           if event.type == pygame.QUIT:
92               run = False
93   
94       if game_state == "start":
95           draw_start_screen()
96           if keys[pygame.K_SPACE]:
97               game_state = "playing"
98   
99       elif game_state == "playing":
100          game_window.fill(black)
101  
102          # Player movement
103          if keys[pygame.K_LEFT] and player_x - player_speed >= 0:
104              player_x -= player_speed
105          if keys[pygame.K_RIGHT] and player_x + player_speed <= window_x - player_size:
106              player_x += player_speed
107          if keys[pygame.K_UP] and player_y - player_speed >= 0:
108              player_y -= player_speed
109          if keys[pygame.K_DOWN] and player_y + player_speed <= window_y - player_size:
110              player_y += player_speed
111  
112          # Draw player
113          pygame.draw.rect(game_window, blue, (player_x, player_y, player_size, player_size))
114  
115          # Draw coin
116          pygame.draw.rect(game_window, yellow, (coin_x, coin_y, coin_size, coin_size))
117  
118          # Move obstacle diagonally
119          obstacle_x += obstacle_speed_x
120          obstacle_y += obstacle_speed_y
121  
122          # Bounce off walls
123          if obstacle_x <= 0 or obstacle_x + obstacle_width >= window_x:
124              obstacle_speed_x *= -1
125          if obstacle_y <= 0 or obstacle_y + obstacle_height >= window_y:
126              obstacle_speed_y *= -1
127  
128          # Draw obstacle
129          pygame.draw.rect(game_window, red, (obstacle_x, obstacle_y, obstacle_width, obstacle_height))
130  
131          # Collision with coin
132          player_rect = pygame.Rect(player_x, player_y, player_size, player_size)
133          coin_rect = pygame.Rect(coin_x, coin_y, coin_size, coin_size)
134          if player_rect.colliderect(coin_rect):
135              score += 10
136              coin_x = random.randrange(0, window_x - coin_size, 20)
137              coin_y = random.randrange(0, window_y - coin_size - 50, 20)
138  
139          # Collision with obstacle
140          obstacle_rect = pygame.Rect(obstacle_x, obstacle_y, obstacle_width, obstacle_height)
141          if player_rect.colliderect(obstacle_rect):
142              lives -= 1
143              player_x = window_x//2
144              player_y = window_y - player_size - 10
145              time.sleep(0.5)
146  
147          # Draw score and lives
148          score_text = font.render("Score: " + str(score), True, white)
149          lives_text = font.render("Lives: " + str(lives), True, white)
150          game_window.blit(score_text, (10,10))
151          game_window.blit(lives_text, (10,40))
152  
153          # Check game over
154          if lives <= 0:
155              game_state = "game_over"
156  
157          pygame.display.update()
158  
159      elif game_state == "game_over":
160          draw_game_over()
161          if keys[pygame.K_r]:
162              reset_game()
163              game_state = "start"
164  
165  pygame.quit()
